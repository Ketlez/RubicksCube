#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColorCub;

out vec3 ColorCub;
out vec4 Pos;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main()
{
    ColorCub = aColorCub;
    Pos = vec4(aPos, 1);
    gl_Position = projection * view * model * vec4(aPos, 1.0);
   
}
