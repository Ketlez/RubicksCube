#version 330 core
out vec4 FragColor;
in vec2 TexCoord;

in vec3 Pos;

uniform float time;
uniform sampler2D ourTexture;
void main()
{
    vec2 textCor = vec2(TexCoord.x, 1-TexCoord.y );

    float speedtime = time;
    float temp = mod(speedtime, 10);

    float red = ((Pos.y))+temp;

    vec4 color = vec4((sin(red+0.5)+1)/2, (sin(red)+1)/2, 1-(sin(red)+1)/2, 1.0f);

    if(Pos.x*Pos.x+Pos.y*Pos.y<=0.02)
    FragColor = mix(texture(ourTexture, textCor), color, 0.5);
    else
    discard;
}
