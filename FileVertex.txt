#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 aTexCoord;
out vec3 Pos;
out vec2 TexCoord;
uniform float time;
uniform vec2 randomPos;
void main()
{
    
    Pos = aPos;
    Pos = Pos/2;
    float cicleTime = time*10;
    
    float x = Pos.x*cos(cicleTime)-Pos.y*sin(cicleTime);
    float y = Pos.x*sin(cicleTime)+Pos.y*cos(cicleTime);
    

    gl_Position = vec4(x+cos(time*10)/2, y+sin(time*10)/2, aPos.z, 1.0);
    TexCoord = aTexCoord;
}
