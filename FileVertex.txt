#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 aTexCoord;
out vec3 Pos;
out vec2 TexCoord;
uniform float time;
uniform vec2 randomPos;
void main()
{
    
    Pos = aPos;
    float cicleTime = mod(time, 10)/10;
    float x = Pos.x*cos(cicleTime)-Pos.y*sin(cicleTime);
    float y = Pos.x*sin(cicleTime)+Pos.y*cos(cicleTime);
    

    gl_Position = vec4(x+randomPos.x, y+randomPos.y, aPos.z, 1.0);
    TexCoord = aTexCoord;
}
